// hooks/useOnboardingProgress.ts
export function useOnboardingProgress() {
  const [progress, setProgress] = useState<Map<string, boolean>>(new Map());
  
  useEffect(() => {
    const saved = localStorage.getItem('onboarding_progress');
    if (saved) {
      setProgress(new Map(JSON.parse(saved)));
    }
  }, []);
  
  const markComplete = (contentId: string) => {
    const newProgress = new Map(progress);
    newProgress.set(contentId, true);
    setProgress(newProgress);
    localStorage.setItem('onboarding_progress', JSON.stringify([...newProgress]));
  };
  
  const getCompletionRate = () => {
    const total = ONBOARDING_CONTENTS.length;
    const completed = [...progress.values()].filter(Boolean).length;
    return Math.round((completed / total) * 100);
  };
  
  return { progress, markComplete, getCompletionRate };
}
